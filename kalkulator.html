// --- JAVASCRIPT UNTUK KALKULATOR (OPERATOR MATEMATIKA) --- function hitung() { // Mengambil nilai dari input let num1 = parseFloat(document.getElementById('num1').value); let num2 = parseFloat(document.getElementById('num2').value); let operator = document.getElementById('operator').value; let resultElement = document.getElementById('result'); // Validasi input if (isNaN(num1) || isNaN(num2)) { resultElement.innerHTML = "Mohon isi angka dengan benar!"; return; } let hasil = 0; // Menggunakan operator matematika switch (operator) { case '+': hasil = num1 + num2; break; case '-': hasil = num1 - num2; break; case '*': hasil = num1 * num2; break; case '/': if (num2 === 0) { resultElement.innerHTML = "Tidak bisa dibagi dengan nol!"; return; } hasil = num1 / num2; break; default: resultElement.innerHTML = "Pilih operator!"; return; } resultElement.innerHTML = "Hasil: " + hasil; }
